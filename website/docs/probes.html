Writing Custom Probes

Extend RedForge with your own probes by subclassing BaseProbe:

python
from redforge.probes.base import BaseProbe, ProbeResult, Severity

class MyCustomProbe(BaseProbe):
    id = "RF-custom-001"
    owasp_id = "LLM01"
    mitre_atlas = "AML.T0051"
    severity = Severity.HIGH
    description = "My custom attack probe"
    tags = ["custom", "my-org"]

    def payloads(self) -> list[str]:
        return [
            "Payload 1: ignore instructions...",
            "Payload 2: alternate framing...",
        ]

    def score(self, response: str, payload: str) -> ProbeResult:
        # Return score 1.0 = vulnerable, 0.0 = safe
        score = 1.0 if "harmful" in response.lower() else 0.0
        return self._make_result(
            payload, response, score,
            evidence="Custom evidence string"
        )
