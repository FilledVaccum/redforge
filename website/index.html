Quick Start

From install to first scan in under 2 minutes

bash
# Install with your provider
pip install redforge
pip install "redforge[openai]"
pip install "redforge[all-providers]"

export OPENAI_API_KEY="sk-..."

# First scan
redforge scan \
  --provider openai \
  --model gpt-4o \
  --authorization owned \
  --format html \
  --output report.html
bash
docker pull redforge/redforge:latest

docker run --rm \
  -e OPENAI_API_KEY="$OPENAI_API_KEY" \
  -v $(pwd)/results:/results \
  redforge/redforge:latest \
  scan --provider openai --model gpt-4o \
       --authorization owned \
       --format html \
       --output /results/report.html
yaml
# redforge.yaml — auto-discovered in project root
provider: openai
model: gpt-4o
authorization: owned
compliance: [NIST_AI_RMF, EU_AI_ACT]
probes:
  - RF-001-prompt-injection
  - community_prompt_injection_extended
  - my_custom_yaml_probe
output:
  formats: [html, sarif, json]
  directory: ./results
bash
# CLI flags override config values
redforge scan               # auto-discovers redforge.yaml
redforge scan --config ci.yaml  # explicit path
python
import asyncio
from redforge import Scanner, ScanConfig

config = ScanConfig(
    provider="openai", model="gpt-4o",
    authorization="owned",
    compliance=["NIST_AI_RMF", "EU_AI_ACT"],
)
report = asyncio.run(Scanner(config).scan())

print(f"Risk: {report.score.risk_level} ({report.score.risk_score}/10)")
for f in report.critical_findings:
    print(f"  [{f.owasp_id}] {f.probe_id}")
